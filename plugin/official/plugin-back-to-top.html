# @vuepress/plugin-back-to-top

Back-to-top plugin

# Install

yarn add -D @vuepress/plugin-back-to-top@next
# OR npm install -D @vuepress/plugin-back-to-top@next

# Usage

module.exports = {
  plugins: ['@vuepress/back-to-top']
}
